    87→function ManageTPDialog({ open, onClose }: { open: boolean; onClose: () => void }) {
    88→  const { toast } = useToast();
    89→  const qc = useQueryClient();
    90→  const [newTP, setNewTP] = useState("");
    91→  const [deleteTarget, setDeleteTarget] = useState<{ id: string; tahunPelajaran: string } | null>(null);
    92→
    93→  const { data: tpList = [], isLoading } = useQuery({
    94→    queryKey: ["tahun-pelajaran"],
    95→    queryFn: async () => {
    96→      const res = await fetch("/api/tahun-pelajaran");
    97→      if (!res.ok) throw new Error("Gagal memuat");
    98→      return res.json();
    99→    },
   100→  });
   101→
   102→  const addMutation = useMutation({
   103→    mutationFn: async (tp: string) => {
   104→      const res = await fetch("/api/tahun-pelajaran", {
   105→        method: "POST",
   106→        headers: { "Content-Type": "application/json" },
   107→        body: JSON.stringify({ tahunPelajaran: tp }),
   108→      });
   109→      const data = await res.json();
   110→      if (!res.ok) throw new Error(data.error || "Gagal menambah");
   111→      return data;
   112→    },
   113→    onSuccess: () => {
   114→      qc.invalidateQueries({ queryKey: ["tahun-pelajaran"] });
   115→      toast({ title: "Berhasil", description: "Tahun pelajaran berhasil ditambahkan" });
   116→      setNewTP("");
   117→    },
   118→    onError: (err) => toast({ title: "Gagal", description: err.message, variant: "destructive" }),
   119→  });
   120→
   121→  const deleteMutation = useMutation({
   122→    mutationFn: async (id: string) => {
   123→      const res = await fetch("/api/tahun-pelajaran", {
   124→        method: "DELETE",
   125→        headers: { "Content-Type": "application/json" },
   126→        body: JSON.stringify({ id }),
   127→      });
   128→      const data = await res.json();
   129→      if (!res.ok) throw new Error(data.error || "Gagal menghapus");
   130→      return data;
   131→    },
   132→    onSuccess: () => {
   133→      qc.invalidateQueries({ queryKey: ["tahun-pelajaran"] });
   134→      toast({ title: "Berhasil", description: "Tahun pelajaran berhasil dihapus" });
   135→      setDeleteTarget(null);
   136→    },
   137→    onError: (err) => toast({ title: "Gagal", description: err.message, variant: "destructive" }),
   138→  });
   139→
   140→  const handleAdd = () => {
   141→    const val = newTP.trim();
   142→    if (!val) return;
   143→    // Validate format: YYYY/YYYY+1
   144→    const match = val.match(/^(\d{4})\/(\d{4})$/);
   145→    if (!match || parseInt(match[2]) !== parseInt(match[1]) + 1) {
   146→      toast({ title: "Format Salah", description: 'Gunakan format "YYYY/YYYY+1", contoh: "2026/2027"', variant: "destructive" });
   147→      return;
   148→    }
   149→    addMutation.mutate(val);
   150→  };
   151→
   152→  return (
   153→    <>
   154→      <Dialog open={open} onOpenChange={(v) => !v && onClose()}>
   155→        <DialogContent className="sm:max-w-md">
   156→          <DialogHeader>
   157→            <DialogTitle>Kelola Tahun Pelajaran</DialogTitle>
   158→            <DialogDescription>Tambah atau hapus tahun pelajaran yang tersedia di sistem.</DialogDescription>
   159→          </DialogHeader>
   160→
   161→          {/* Add new */}
   162→          <div className="flex gap-2">
   163→            <Input
   164→              placeholder="Contoh: 2026/2027"
   165→              value={newTP}
   166→              onChange={(e) => setNewTP(e.target.value)}
   167→              onKeyDown={(e) => e.key === "Enter" && handleAdd()}
   168→              className="flex-1"
   169→            />
   170→            <Button onClick={handleAdd} disabled={addMutation.isPending || !newTP.trim()} size="sm">
   171→              {addMutation.isPending ? <Loader2 className="w-4 h-4 animate-spin" /> : <Plus className="w-4 h-4" />}
   172→              Tambah
   173→            </Button>
   174→          </div>
   175→
   176→          {/* List */}
   177→          <div className="max-h-64 overflow-y-auto space-y-1">
   178→            {isLoading && (
   179→              <div className="flex items-center justify-center py-8">
   180→                <Loader2 className="w-5 h-5 animate-spin text-muted-foreground" />
   181→              </div>
   182→            )}
   183→            {!isLoading && tpList.length === 0 && (
   184→              <p className="text-sm text-muted-foreground text-center py-6">Belum ada data.</p>
   185→            )}
   186→            {tpList.map((tp: { id: string; tahunPelajaran: string }) => (
   187→              <div
   188→                key={tp.id}
   189→                className="flex items-center justify-between px-3 py-2 rounded-md border border-border hover:bg-muted/50"
   190→              >
   191→                <span className="text-sm font-medium">{tp.tahunPelajaran}</span>
   192→                <Button
   193→                  variant="ghost"
   194→                  size="icon"
   195→                  className="h-7 w-7 text-destructive hover:text-destructive"
   196→                  onClick={() => setDeleteTarget(tp)}
   197→                >
   198→                  <Trash2 className="w-3.5 h-3.5" />
   199→                </Button>
   200→              </div>
   201→            ))}
   202→          </div>
   203→        </DialogContent>
   204→      </Dialog>
   205→
   206→      {/* Delete confirmation */}
   207→      <AlertDialog open={!!deleteTarget} onOpenChange={(v) => !v && setDeleteTarget(null)}>
   208→        <AlertDialogContent>
   209→          <AlertDialogHeader>
   210→            <AlertDialogTitle>Hapus Tahun Pelajaran?</AlertDialogTitle>
   211→            <AlertDialogDescription>
   212→              Yakin ingin menghapus <strong>{deleteTarget?.tahunPelajaran}</strong>? Data siswa, guru, dan mutasi pada tahun pelajaran ini tetap ada di database.
   213→            </AlertDialogDescription>
   214→          </AlertDialogHeader>
   215→          <AlertDialogFooter>
   216→            <AlertDialogCancel>Batal</AlertDialogCancel>
   217→            <AlertDialogAction
   218→              onClick={() => deleteTarget && deleteMutation.mutate(deleteTarget.id)}
   219→              disabled={deleteMutation.isPending}
   220→              className="bg-destructive text-destructive-foreground hover:bg-destructive/90"
   221→            >
   222→              {deleteMutation.isPending ? <Loader2 className="w-4 h-4 animate-spin" /> : null}
   223→              Hapus
   224→            </AlertDialogAction>
   225→          </AlertDialogFooter>
   226→        </AlertDialogContent>
   227→      </AlertDialog>
   228→    </>
   229→  );
   230→}

   234→function TahunPelajaranSelector({ onManage }: { onManage: () => void }) {
   235→  const { tahunPelajaran, setTahunPelajaran, semester, setSemester } = useAppStore();
   236→
   237→  const { data: tpList = [] } = useQuery({
   238→    queryKey: ["tahun-pelajaran"],
   239→    queryFn: async () => {
   240→      const res = await fetch("/api/tahun-pelajaran");
   241→      if (!res.ok) return [];
   242→      return res.json();
   243→    },
   244→  });
   245→
   246→  // If API data loaded, ensure the current selection is valid
   247→  const options = tpList.map((tp: { tahunPelajaran: string }) => tp.tahunPelajaran);
   248→  const isValid = options.includes(tahunPelajaran);
   249→
   250→  // Auto-sync store when the current tahunPelajaran is not in the API list
   251→  useEffect(() => {
   252→    if (!isValid && options.length > 0) {
   253→      setTahunPelajaran(options[0]);
   254→    }
   255→  }, [isValid, options, setTahunPelajaran]);
   256→
   257→  return (
   258→    <div className="space-y-3">
   259→      <div className="flex items-center justify-between px-1">
   260→        <div className="flex items-center gap-2">
   261→          <CalendarDays className="w-3.5 h-3.5 text-white/40 flex-shrink-0" />
   262→          <span className="text-[10px] font-bold uppercase tracking-[0.12em] text-white/40">
   263→            Tahun Pelajaran
   264→          </span>
   265→        </div>
   266→        <button
   267→          onClick={onManage}
   268→          className="p-1 rounded-md hover:bg-white/10 text-white/30 hover:text-white/60 transition-colors"
   269→          title="Kelola Tahun Pelajaran"
   270→        >
   271→          <Settings className="w-3.5 h-3.5" />
   272→        </button>
   273→      </div>
   274→      <div className="px-1 space-y-2">
   275→        <div className="space-y-1.5">
   276→          <Label className="text-[11px] text-white/35 font-medium">Tahun Ajaran</Label>
   277→          <Select value={isValid ? tahunPelajaran : options[0] || ""} onValueChange={setTahunPelajaran}>
   278→            <SelectTrigger className="w-full h-9 text-sm bg-white/8 border-white/10 text-white/80 hover:bg-white/12 focus:ring-white/20 data-[placeholder]:text-white/30">
   279→              <SelectValue placeholder={options.length === 0 ? "Belum ada data" : "Pilih..."} />
   280→            </SelectTrigger>
   281→            <SelectContent>
   282→              {options.map((tp: string) => (
   283→                <SelectItem key={tp} value={tp} className="text-sm">
   284→                  {tp}
   285→                </SelectItem>
   286→              ))}
   287→            </SelectContent>
   288→          </Select>
   289→        </div>
   290→        <div className="space-y-1.5">
   291→          <Label className="text-[11px] text-white/35 font-medium">Semester</Label>
   292→          <Select value={semester} onValueChange={setSemester}>
   293→            <SelectTrigger className="w-full h-9 text-sm bg-white/8 border-white/10 text-white/80 hover:bg-white/12 focus:ring-white/20">
   294→              <SelectValue />
   295→            </SelectTrigger>
   296→            <SelectContent>
   297→              {SEMESTER_OPTIONS.map((s) => (
   298→                <SelectItem key={s} value={s} className="text-sm">
   299→                  Semester {s}
   300→                </SelectItem>
   301→              ))}
   302→            </SelectContent>
   303→          </Select>
   304→        </div>
   305→      </div>
   306→    </div>
   307→  );
   308→}

   312→function Sidebar({ open, onClose, collapsed, onToggleCollapse, className }: { open: boolean; onClose: () => void; collapsed: boolean; onToggleCollapse: () => void; className?: string }) {
   313→  const { activePage, setActivePage } = useAppStore();
   314→  const [manageTPOpen, setManageTPOpen] = useState(false);
   315→  const { data: session } = useSession();
   316→  const { toast } = useToast();
   317→
   318→  // Change password state
   319→  const [cpOpen, setCpOpen] = useState(false);
   320→  const [cpOld, setCpOld] = useState('');
   321→  const [cpNew, setCpNew] = useState('');
   322→  const [cpConfirm, setCpConfirm] = useState('');
   323→  const [cpErrors, setCpErrors] = useState<Record<string, string>>({});
   324→  const [cpLoading, setCpLoading] = useState(false);
   325→  const [showOld, setShowOld] = useState(false);
   326→  const [showNew, setShowNew] = useState(false);
   327→  const [showConfirm, setShowConfirm] = useState(false);
   328→
   329→  const handleChangePassword = async () => {
   330→    const errors: Record<string, string> = {};
   331→    if (!cpOld.trim()) errors.oldPassword = 'Password lama wajib diisi';
   332→    if (!cpNew.trim()) errors.newPassword = 'Password baru wajib diisi';
   333→    else if (cpNew.length < 6) errors.newPassword = 'Password baru minimal 6 karakter';
   334→    if (!cpConfirm.trim()) errors.confirmNewPassword = 'Konfirmasi password wajib diisi';
   335→    else if (cpNew !== cpConfirm) errors.confirmNewPassword = 'Konfirmasi password tidak cocok';
   336→    if (Object.keys(errors).length > 0) { setCpErrors(errors); return; }
   337→
   338→    setCpLoading(true);
   339→    setCpErrors({});
   340→    try {
   341→      const userId = (session?.user as { id?: string })?.id;
   342→      const res = await fetch('/api/user/change-password', {
   343→        method: 'POST',
   344→        headers: { 'Content-Type': 'application/json' },
   345→        body: JSON.stringify({ userId, oldPassword: cpOld, newPassword: cpNew, confirmNewPassword: cpConfirm }),
   346→      });
   347→      const data = await res.json();
   348→      if (!res.ok) {
   349→        const field = data.error?.includes('lama') ? 'oldPassword' : 'newPassword';
   350→        setCpErrors({ [field]: data.error });
   351→        return;
   352→      }
   353→      toast({ title: 'Berhasil', description: 'Password berhasil diubah' });
   354→      setCpOpen(false);
   355→      setCpOld(''); setCpNew(''); setCpConfirm('');
   356→    } catch {
   357→      toast({ title: 'Gagal', description: 'Terjadi kesalahan', variant: 'destructive' });
   358→    } finally {
   359→      setCpLoading(false);
   360→    }
   361→  };
   362→
   363→  const closeCp = () => { setCpOpen(false); setCpOld(''); setCpNew(''); setCpConfirm(''); setCpErrors({}); };
   364→
   365→  return (
   366→    <>
   367→      {open && (
   368→        <div
   369→          className="fixed inset-0 z-40 bg-black/40 lg:hidden"
   370→          onClick={onClose}
   371→        />
   372→      )}
   373→
   374→      <aside
   375→        className={cn(
   376→          "fixed top-0 left-0 z-50 h-full w-64 flex flex-col transition-all duration-300 ease-in-out",
   377→          "[background:linear-gradient(180deg,oklch(0.17_0.04_160)_0%,oklch(0.14_0.05_170)_50%,oklch(0.12_0.04_260)_100%)]",
   378→          open && !collapsed ? "translate-x-0" : "-translate-x-full",
   379→          !collapsed && "lg:translate-x-0 lg:static lg:z-auto",
   380→          collapsed && "lg:-translate-x-full lg:absolute lg:z-auto",
   381→          className
   382→        )}
   383→      >
   384→        {/* Header */}
   385→        <div className="flex items-center gap-3 px-5 py-5 border-b border-white/10">
   386→          <div className="flex items-center justify-center w-9 h-9 rounded-xl bg-white/15 backdrop-blur-sm text-white shadow-lg shadow-emerald-900/20">
   387→            <School className="w-5 h-5" />
   388→          </div>
   389→          <div className="flex-1 min-w-0">
   390→            <h1 className="text-sm font-bold text-white leading-tight">Operator Sekolah</h1>
   391→            <p className="text-[11px] text-white/50 leading-tight">Sistem Informasi</p>

   394→            onClick={onClose}
   395→            className="lg:hidden p-1.5 rounded-lg hover:bg-white/10 text-white/70 hover:text-white transition-colors"

   397→            <X className="w-4 h-4" />
   398→          </button>

   400→            onClick={onToggleCollapse}
   401→            className="hidden lg:flex p-1.5 rounded-lg hover:bg-white/10 text-white/50 hover:text-white transition-colors"

   404→            <PanelLeftClose className="w-4 h-4" />
   405→          </button>
   406→        </div>
   407→
   408→        {/* Tahun Pelajaran Selector */}
