   742тЖТ   742тЖТfunction generatePageNumbers(
   743тЖТ   743тЖТ  current: number,
   744тЖТ   744тЖТ  total: number
   745тЖТ   745тЖТ): (number | string)[] {
   746тЖТ   746тЖТ  if (total <= 7) {
   747тЖТ   747тЖТ    return Array.from({ length: total }, (_, i) => i + 1);
   748тЖТ   748тЖТ  }
   749тЖТ   749тЖТ
   750тЖТ   750тЖТ  const pages: (number | string)[] = [1];
   751тЖТ   751тЖТ
   752тЖТ   752тЖТ  if (current > 3) {

   754тЖТ   754тЖТ  }
   755тЖТ   755тЖТ
   756тЖТ   756тЖТ  const start = Math.max(2, current - 1);

   758тЖТ   758тЖТ
   759тЖТ   759тЖТ  for (let i = start; i <= end; i++) {

   762тЖТ   762тЖТ
   763тЖТ   763тЖТ  if (current < total - 2) {
   764тЖТ   764тЖТ    pages.push('...');
   765тЖТ   765тЖТ  }
   766тЖТ   766тЖТ
   767тЖТ   767тЖТ  pages.push(total);
   768тЖТ   768тЖТ
   769тЖТ   769тЖТ  return pages;
